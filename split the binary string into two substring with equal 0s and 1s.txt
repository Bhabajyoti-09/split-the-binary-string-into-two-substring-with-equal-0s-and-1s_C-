class Solution {
public:
    int maxSubStr(string str){
        //Write your code here
        int count =0,one=0,zero=0;{
        for(int i=0;i<str.size();i++){
            if(str[i]=='0'){
                zero++;}
                else
            one++;
                if(zero==one){
                count++;}
            }
        }
            if(zero==one){
            return count;}
            else 
            return -1;
        }
    
};
